apt install shadowsocks-libev shadowsocks-v2ray-plugin nginx

systemctl status shadowsocks-libev

systemctl restart nginx
systemctl status nginx
systemctl stop nginx

#lsof -i :8388 无法显示端口占用情况，取消动态用户
/lib/systemd/system/shadowsocks-libev.service
DynamicUser=false

vi /etc/shadowsocks-libev/config.json

{
    "server":["::0", "0.0.0.0"],
    "mode":"tcp_and_udp",
    "server_port":8388,
    "local_port":1080,
    "password":"passwdsec",
    "timeout":60,
    "method":"chacha20-ietf-poly1305",
    "plugin":"ss-v2ray-plugin",
    "plugin_opts":"server;path=/wow;loglevel=none"
}

#单独使用shadowsocks-libev+WebSocket+TLS,plugin_opts选项使用如下配置
"plugin_opts":"server;path=/wow;loglevel=none;tls;host=ipv6.wower.cool;cert=/etc/ssl/certs/ipv6.wower.cool.pem;key=/etc/ssl/private/ipv6.wower.cool.key"

#QUIC模式(测试不成功）
"mode":"tcp_only"
